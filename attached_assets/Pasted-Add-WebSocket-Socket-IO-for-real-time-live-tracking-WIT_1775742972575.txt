### 1. Install

npm install socket.io

### 2. Update server (main index.ts / server.ts)

Add:

```ts
import { Server } from "socket.io";

let io: Server;

export function initSocket(server: any) {
  io = new Server(server, {
    cors: {
      origin: "*",
    },
  });

  io.on("connection", (socket) => {
    console.log("Client connected:", socket.id);
  });
}

export function emitLocationUpdate(data: any) {
  if (io) {
    io.emit("location:update", data);
  }
}
```

### 3. Initialize socket

After creating HTTP server:

```ts

  console.log(`Server running`);
```

```ts
function getStatus(lastSeen: Date, speed: number) {
  const now = Date.now();
  const diff = now - new Date(lastSeen).getTime();

  if (diff > 2 * 60 * 1000) return "OFFLINE";
  if (speed > 5) return "MOVING";
  if (speed === 0) return "IDLE";

  return "ONLINE";
}
```
